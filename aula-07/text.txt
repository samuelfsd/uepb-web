// entendendo o estado

export function App() {
  let counter = 0

  const add = () => {
    counter += 1
    console.log(counter)
  }

  const remove = () => {
    counter -= 1
    console.log(counter)
  }

  return (
    <div>
      <div>
        {counter !== 0 && <span>{counter}</span>}
      </div>
      <button onClick={add}>+</button>
      <button onClick={remove}>-</button>
    </div>
  )
}
